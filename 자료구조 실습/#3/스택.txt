#include <stdio.h>
#define MAX_SIZE 10

int stack[MAX_SIZE], top = -1;

void push(int);
int pop();
void print_stack();

void main(void)
{
	int menu;
	while (1)
	{
		printf("push(>0), pop(0), exit(-1):");
		scanf("%d", &menu);
		if (menu > 0) {
			push(menu);
			printf("%d push 후: ", menu);
			print_stack();
		}
		else if (menu == 0)
		{
			printf("%d를 pop한 후: ", pop());
			print_stack();
		}
		else
			return;
	}
}

void push(int data)
{
	if (top == MAX_SIZE - 1)
	{
		printf("Stack Full!!\n");
		return;
	}
	stack[++top] = data;
}

int pop()
{
	if(top < 0)
	{
		printf("Stack Empty!!\n");
		return;
	}
	return stack[top--];
}

void print_stack()
{
	printf("[ ");
	for (int i = 0; i <= top; i++)
		printf("%d ", stack[i]);
	printf("], top = %d\n", top);
}
